22TEST_CASE("Configure a module", "[Configurator, ModuleLoader]")
23{
24 Configurator::clear();
25
26 // Create the fake command line, selecting the Impl1 implementation
27 ArgV argvee({"cmtest", "--Modules.ITest=Impl1"});
28
29 Configurator::setCommandLine(argvee.argc(), argvee());

31 ConfiguredModule::parseConfigurator();

54TEST_CASE("Don't configure a module from a stream", "[Configurator, ModuleLoader]")

64 std::unique_ptr<std::istream> pcstream(new std::stringstream(config.str()));
65 Configurator::addStream(std::move(pcstream));

67 // Parse the available modules. This should not change the implementation
