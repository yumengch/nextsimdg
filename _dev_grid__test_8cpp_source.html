24TEST_CASE("Write out a DevGrid restart file", "[DevGrid]")
25{
26 ModuleLoader::getLoader().setAllDefaults();

28 DevGrid grid;

30 grid.setIO(new DevGridIO(grid));
31 // Fill in the data. It is not real data.
32 grid.resetCursor();
33 int nx = DevGrid::nx;
34 int ny = DevGrid::nx;
35 double yFactor = 0.01;
36 double xFactor = 0.0001;
37
38 for (int j = 0; j < ny; ++j) {
39 for (int i = 0; i < nx; ++i) {
40 if (grid.validCursor()) {
41 double fractional = j * yFactor + i * xFactor;
42 grid.cursorData() = PrognosticGenerator()
43 .hice(1 + fractional)
44 .cice(2 + fractional)
45 .sst(3 + fractional)
46 .sss(4 + fractional)
47 .hsnow(5 + fractional)
48 .tice({ -(1. + fractional) });
49 grid.incrCursor();
50 }
51 }
52 }

54 grid.dump(filename);

56 DevGrid grid2;
57 grid2.init("");
58 grid2.setIO(new DevGridIO(grid2));
59
60 grid2.cursor = 0;
61
62 for (int j = 0; j < ny; ++j) {
63 for (int i = 0; i < nx; ++i) {
64 if (grid2.cursor) {

66 ++grid2.cursor;
