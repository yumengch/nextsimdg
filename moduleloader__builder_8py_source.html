3def denamespace(nname):
4 """Returns the last element of the name, without any of the qualifying
5 namespaces."""
6 return nname.split(":")[-1]

8def get_pname(full_name):
9 """Returns the name of the pointer to the stored instance of the
10 implementing class, given the full name of that class."""
11 return f"p_{denamespace(full_name)}"
12
13def get_iname(full_name):
14 """Returns the name of the stored implementation."""
15 return f"i_{denamespace(full_name)}"

17def get_pfname(full_name):
18 """Returns the name of the function pointer for the interface, given its
19 namespaced class name."""
20 return f"pf_{denamespace(full_name)}"

22def get_fname(impl):
23 """Returns the function name for the implementation, given its namespaced string name."""
24 return f"new{denamespace(impl)}"

26def headers(all_implementations, ipp_prefix, hpp_prefix):
27 """Generates the moduleLoaderHeaders.ipp file."""
28 with open(f"{ipp_prefix}moduleLoaderHeaders.ipp", "w", encoding="utf-8") as fil:
29 for interface in all_implementations:
30 header_name = denamespace(interface["name"])
31 fil.write(f"#include \"{hpp_prefix}{header_name}.hpp\"\n")
32 for impl in interface["implementations"]:
33 header_name = denamespace(impl)
34 fil.write(f"#include \"{hpp_prefix}{header_name}.hpp\"\n")
35 # An extra line between interfaces
36 fil.write("\n")

38def functions(all_implementations, ipp_prefix):
39 """Generates the moduleLoaderFunctions.ipp file."""
40 with open(f"{ipp_prefix}moduleLoaderFunctions.ipp", "w", encoding="utf-8") as fil:
41 for interface in all_implementations:
42 name = interface["name"]
43 # Define the pointer to the stored implementation
44 p_name = get_pname(name)
45 fil.write(f"static {name}* {p_name};\n")
46 # Define the function that returns the pointer to the stored implementation
47 fil.write(
48 "template<>\n"
49 f"{name}& ModuleLoader::getImplementation()\n"
50 "{\n"
51 f" return *{p_name};\n"
52 "}\n"
53 )
54 # Define the pointer to function
55 pf_name = get_pfname(name)
56 fil.write(f"std::unique_ptr<{name}> (*{pf_name})();\n")
57 # Define function that call the function pointer
58 fil.write(
59 "template<>\n"
60 f"std::unique_ptr<{name}> ModuleLoader::getInstance() const\n"
61 "{\n"
62 f" return (*{pf_name})();\n"
63 "}\n"
64 )
65 for impl in interface["implementations"]:
66 # The stored instance of the implementation
67 fil.write(f"static {impl} {get_iname(impl)};\n")
68 # The function that return the new unique_ptr for each implementation
69 fil.write(
70 f"std::unique_ptr<{name}> {get_fname(impl)}()\n"
71 "{\n"
72 f" return std::unique_ptr<{impl}>(new {impl});\n"
73 "}\n"
74 )
75 # An extra line between interfaces
76 fil.write("\n")
