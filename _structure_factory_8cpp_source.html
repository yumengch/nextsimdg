20std::shared_ptr<IStructure> StructureFactory::generate(const std::string& structureName)
21{
22 ModuleLoader& loader = ModuleLoader::getLoader();
23 std::string iStruct = "Nextsim::IStructure";
24 std::shared_ptr<IStructure> shst;
25 for (auto struc : loader.listImplementations(iStruct)) {
26 loader.setImplementation(iStruct, struc);
27 if (loader.getImplementation<IStructure>().structureType() == structureName) {
28 shst = std::move(loader.getInstance<IStructure>());
29
30 // TODO There must be a better way
31 if (shst->structureTypeCheck(DevGrid::structureName)) {
32 std::shared_ptr<DevGrid> shdg = std::dynamic_pointer_cast<DevGrid>(shst);
33 shdg->setIO(new DevGridIO(*shdg));
34 }
35
36 return shst;
37 }
38 }
39 // If we reach here, then no valid handlers of the named structure were
40 // found. Throw a invalid argument exception.
41 std::string what = "StructureSelector: Invalid structure name (";
42 what += structureName + ") provided.";
43 throw std::invalid_argument(what);
44}

46std::shared_ptr<IStructure> StructureFactory::generateFromFile(const std::string& filePath)
47{
48 netCDF::NcFile ncf(filePath, netCDF::NcFile::read);
49 netCDF::NcGroup metaGroup(ncf.getGroup(IStructure::metadataNodeName()));
50 netCDF::NcGroupAtt att = metaGroup.getAtt(IStructure::typeNodeName());
51 int len = att.getAttLength();
52 // Initialize a std::string of len, filled with zeros
53 std::string structureName(len, '\0');
54 // &str[0] gives access to the buffer, guaranteed by C++11
