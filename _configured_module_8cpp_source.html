19void ConfiguredModule::parseConfigurator()
20{
21 // A default string that can never be a valid C++ class name
22 std::string defaultStr = "+++DEFAULT+++";
23 // Construct a new options map
24 boost::program_options::options_description opt;
25
26 ModuleLoader& loader = ModuleLoader::getLoader();
27 for (const std::string& module : loader.listModules()) {
28 std::string defaultImpl = *loader.listImplementations(module).begin();
29 opt.add_options()(addPrefix(module).c_str(),
30 boost::program_options::value<std::string>()->default_value(defaultStr),
31 ("Load an implementation of " + module).c_str());
32 }
33
34 boost::program_options::variables_map vm = Configurator::parse(opt);
35
36 for (const std::string& module : loader.listModules()) {
37 std::string implString = vm[addPrefix(module)].as<std::string>();
38 // Only do anything if the retrieved option is not the default value
39 if (implString != defaultStr) {
40 // Check that the retrieved value is one of the implementations
41 // defined for this module
42 std::string moduleImpl = "";
43 for (const std::string implName : loader.listImplementations(module)) {
44 if (implString == implName)
45 moduleImpl = implName;
46 }
47 if (moduleImpl != "") {
48 loader.setImplementation(module, implString);
49 } else {
50 std::string what = "Invalid implementation \"";
51 what += implString + "\" of module " + module + ".";
52 throw std::domain_error(what);
53 }
54 }
55 }
56}
